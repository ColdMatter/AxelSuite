267  public int[] dist2steps(double dist)
268  {
269  int[] rt = new int[2];
270  double nat = dist / step2mmSlope - step2mmIntercept;
271  bool negative = nat < 0;
272  nat = Math.Abs(nat);
273  rt[0] = (int)Math.Floor(nat); if (negative) rt[0] = -rt[0];
274  rt[1] = (int)Math.Floor((nat % 1) * 256.0); if (negative) rt[1] = -rt[1];
275  return rt;
